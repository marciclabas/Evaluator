84 void User::parseSubmission(prb::ID problemID, prb::result r) {
85  solvableProblems.addSubmission(problemID);
86  stats.totalSubmissions++;
87  // if solved for the first time, update total problems tried
88  if(solvableProblems.submissionsCount(problemID) == 1) stats.triedProblems++;
89 
90  if(r == prb::result::accepted) {

92  solvedProblems.addProblem(problemID, solvableProblems.submissionsCount(problemID));
